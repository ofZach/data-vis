    var hpBulkSearch = {
        xrequest: null,
        clearTb: true,
        errorMessage: 'Enter valid domain names to search.\nDomain names should be 63 characters or less\nand should contain only letters, numbers or hyphens.',
        errorCount: 0,
        search: function (searchBox, tldselector, searchtype, scrollTo) {
            hpBulkSearch.errorCount = 0;
            var search$ = $(searchBox),
                domains = this._domfilter(search$.val().split('\n')),
                tlds = [];

            $(tldselector).each(function () { tlds.push($(this).val()); });
            if (search$.data('clear') || search$.val().replace(/^\s+|\s+$/g, '') === '') {
                alertBox.showAlertErrorBox('Please enter at least one domain');
                return;
            }
            
            if (hpBulkSearch.errorCount === 0) {
                    if (this._singleSearch(domains, tlds)) {
                        var url = 'http://www.godaddy.com/domains/search.aspx',
                                    hdnOffer = $('#hdnfbOfferId'),
                                    offerId = (hdnOffer.val()) ? '&offer_id=' + hdnOffer.val() : '',
                                    tld = (domains[0].indexOf('.') !== -1)? domains[0].substring(domains[0].indexOf('.') + 1): (tlds[0])? tlds[0]: 'com';

                    }
                    else{
                                        
                        this.submitBulkSearch(domains, tlds, searchtype);
                    
                    }
            }
            else { alertBox.showAlertErrorBox(hpBulkSearch.errorMessage); }           
        },
        submitBulkSearch: function(domains, tlds, searchtype){
          bsmodal.display('bulkSearchContentSearching');
          hdnOffer = $('#hdnfbOfferId'),
          offerId = (hdnOffer.val()) ? '&offer_id=' + hdnOffer.val() : '',
          this.xrequest = $.ajax({
              searchCount: '1_GDHP_BulkSearchCount',
              type: 'POST',
              url: 'http://www.godaddy.com/shared/homepage2/1/actions/express-buy.aspx',
              data: "domains=" + domains.join('|') + "&dottypes=" + tlds.join('|') + "&type=" + searchtype + "&searchCount=1_GDHP_BulkSearchCount&sourcepage=gdhp-BulkSearch.BulkDomainSearch" + offerId,
              success: function (data) { hpBulkSearch._redirect(data) }
          });
        },
        _singleSearch: function(domains, tlds){
            var singleSearch = false,
                searchSplit = false;

            if(!searchSplit){
                if (domains.length === 1) {
                    if (domains[0].indexOf('.') !== -1) {
                        var currentTld = domains[0].substring(domains[0].indexOf('.') + 1);
                        if (tlds.length){
                            singleSearch = (currentTld === tlds[0]);
                        }
                        else{
                            singleSearch = true;
                        }
                    }
                    else{
                     singleSearch = (tlds.length <= 1);
                    }
                }
            }
            return singleSearch;
        },
         _domfilter: function (array) {
            var result = [],
                limit = 500;
            for (var i = 0; i < array.length; i++) {
                if (i <= limit - 1) {// if not greater then limit
                     var dom = escape(array[i].replace('|', '').replace(/^\s+|\s+$/g, ''));
                     if ($.inArray(dom, result) === -1 && dom !== '') { //does not already exist in array
                        (this._isValid(dom))? result.push(dom) : hpBulkSearch.errorCount++;
                     }
                }
            }
            return result;
        },
        _isValid: function(domainName) { 
            var regExInvalidChars = /[^a-zA-Z0-9-\s.]+/,
                regExBeginsWithPeriodOrDash = /^[\.|-]+/,
                domLength = (domainName.length > 63),                     
                invalidChar = (regExInvalidChars.test(domainName)),
                beginswith = (regExBeginsWithPeriodOrDash.test(domainName));
            return (!domLength && !invalidChar && !beginswith);       
        },
        _displayerror: function () {

<script type="text/javascript" id="jsonContent" language="javascript">
  var jsonContent = {
    load: function (targetDivId, url, async) {
      if ($("#" + targetDivId).html().length == 0) {
        if (url != null) {
          url += (url.indexOf("?") >= 0) ? "&" : "?";
          url += "callback=jsonContent._fill&targetDivId=" + targetDivId;
          jsonContent._call("jsonContent._fill", url, async);
        }
      }
      else {
        this._fill({ TargetDivID: targetDivId });
      }
    },
    _call: function (f, u, async) {
      $.ajax({ dataType: "jsonp", jsonp: f, url: u, async: async });
    },
    _fill: function (data) {
      if (data != null) {
        var targetDivId = data.TargetDivID;
        if (data.Html != null) { $("#" + targetDivId).html(data.Html); }
      }
    }
  }
</script>

      if (gdhp_bobsblog_quote_ticker.currentQuoteIndex > -1) {
        gdhp_bobsblog_quote_ticker._cancelSlideShow();
      }
    },

    _cancelSlideShow: function () {
      if (typeof gdhp_bobsblog_quote_ticker.timeoutId == "number") {
        window.clearTimeout(gdhp_bobsblog_quote_ticker.timeoutId);
        delete gdhp_bobsblog_quote_ticker.timeoutId;
      }
    },

    _showQuoteForShowDuration: function () {
      gdhp_bobsblog_quote_ticker._cancelSlideShow();
      gdhp_bobsblog_quote_ticker._showCurrentQuote();
      gdhp_bobsblog_quote_ticker.timeoutId = window.setTimeout('gdhp_bobsblog_quote_ticker._showNextQuote()', gdhp_bobsblog_quote_ticker.showDuration);
    },

    _showCurrentQuote: function () {
      var currentQuote = gdhp_bobsblog_quote_ticker.quotes[gdhp_bobsblog_quote_ticker.currentQuoteIndex];
      if (gdhp_bobsblog_quote_ticker.tickerDisplayElement.innerHTML != currentQuote) {
        gdhp_bobsblog_quote_ticker.tickerDisplayElement.innerHTML = currentQuote;
      }
    },

    _showNextQuote: function () {
      delete gdhp_bobsblog_quote_ticker.timeoutId;
      gdhp_bobsblog_quote_ticker.currentQuoteIndex = (gdhp_bobsblog_quote_ticker.currentQuoteIndex + 1) % gdhp_bobsblog_quote_ticker.quotes.length;
